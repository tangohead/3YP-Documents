12  public class SmallWorld {
13      /**
14       * Constructs a directed small-world graph. Yes, that's not what the Watts & Strogatz Model is supposed to do, but oh well.
15       * @param context
16       * @param networkName
17       * @param numNodes
18       * @param SWBeta
19       * @param SWDegree
20       * @return
21       */
22      public static Network generateRSW(Context context, String networkName, int numNodes, double SWBeta, int SWDegree)
23      {
24          //Add the nodes to the network
25          for( int i = 0; i < numNodes ; i++) {
26              context.add(new BaseHuman("n" + i, context));
27          }
28          NetworkBuilder<BaseHuman> builder = new NetworkBuilder(networkName, context, true);
29          NetworkGenerator gen = new WattsBetaSmallWorldGenerator(0.2, 2, false);
30          
31          
32          builder.setGenerator(gen);
33          Network<BaseHuman> network = builder.buildNetwork();
34          
35          Iterable<BaseHuman> nodes = network.getNodes();
36          Iterator<BaseHuman> iter = nodes.iterator();
37          while(iter.hasNext())
38          {
39              iter.next().setNetwork(network);
40          }
41          
42          return network;
43      }
