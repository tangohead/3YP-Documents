17      /**
18       * Adds a BaseHuman class to a Repast Network
19       * @param humanInstance - the instance to be added
20       * @param context - Repast context
21       * @param network - Repast network
22       * @param reciprocate - whether to automatically reciprocate with the node that this instance connects to.
23       * @param reciprocateChance - the chance of an edge being reciprocated
24       */
25      public static void addToRSF(BaseHuman humanInstance, Context<?> context, Network<BaseHuman> network, boolean reciprocate, double reciprocateChance, NDParams nd)
26      {
27          //Add to scale-free network
28          Iterable<BaseHuman> nodes = network.getNodes();
29  
30          ArrayList<ArrayList<BaseHuman>> newEdges = new ArrayList<ArrayList<BaseHuman>>();
31          for(BaseHuman current : nodes)
32          {
33              double iDegree = network.getDegree(current);
34              double jSumDegree = 0;
35              for(BaseHuman n : network.getNodes())
36              {
37                  jSumDegree += network.getDegree(n);
38              }
39              
40              double prob = iDegree / jSumDegree;
41              prob = prob * humanInstance.compatProb(current);
42              /*if(this.isSmoker() != current.isSmoker())
43                  prob = prob * 0.5;*/
44              
45              if(Math.random() <= prob)
46              {
47                  ArrayList<BaseHuman> pair = new ArrayList<BaseHuman>();
48                  pair.add(humanInstance);
49                  pair.add(current);
50                  newEdges.add(pair);
51                  
52                  if(reciprocate && Math.random() <= reciprocateChance)
53                  {
54                      ArrayList<BaseHuman> rPair = new ArrayList<BaseHuman>();
55                      rPair.add(current);
56                      rPair.add(humanInstance);
57                      newEdges.add(rPair);
58                  }
59              }
60          }
61          
62          for(ArrayList<BaseHuman> pair : newEdges)
63          {
64              //check if we have NDParams, if not, no weights.
65              //System.out.println("Adding an edge...");
66              if(nd == null)
67                  network.addEdge(pair.get(0), pair.get(1));
68              else
69                  network.addEdge(pair.get(0), pair.get(1), Distributions.getND(nd));
70          }
71      }

73      /**
74       * 
75       * @param context
76       * @param networkName name of the network to be created
77       * @param startWith number of nodes to start the network with
78       * @param numNodes number of nodes to add to the network
79       * @param reciprocate whether to reciprocate network links (wrt reciprocate chance)
80       * @param reciprocateChance the chance of reciprocating an edge
81       * @return
82       */
83      public static Network<BaseHuman> createRSF(Context<Object> context, 
84              String networkName, int startWith, int numNodes, 
85              boolean reciprocate, double reciprocateChance, boolean removeOutliers,
86              NDParams nd)
87      {
88          NetworkBuilder<BaseHuman> builder = new NetworkBuilder(networkName, context, true);
89          Network<BaseHuman> network = builder.buildNetwork();
90          
91          //First up, we generate a random starting network
92          //Add the initial nodes (must be more than 2)
93          if(startWith < 2)
94              startWith = 2;
95          for(int i = 0; i < startWith; i++)
96          {
97              //note params don't matter since this is the base
98              context.add(new BaseHuman("n" + i, context, network, false, false, 0, null));
99          }
100         
101         network = buildSFFoundation(context, network, nd);
102 
103         //Now add the rest onto the base network
104         for( int i = startWith; i < numNodes ; i ++) {
105             context.add(new BaseHuman("n" + i, context, network, true, reciprocate, reciprocateChance, nd));
106         }
107         
108         return network;
109     }

112     public static Network<BaseHuman> SFOnBase(Context<Object> context, 
113             Network<BaseHuman> currNet, int numNodes, 
114             boolean reciprocate, double reciprocateChance, boolean removeOutliers,
115             NDParams nd)
116     {
117         
118         Network<BaseHuman> network = buildSFFoundation(context, currNet, nd);
119         
120         //Now add the rest onto the base network
121         for( int i = context.getObjects(BaseHuman.class).size(); i < numNodes ; i ++) {
122             context.add(new BaseHuman("n" + i, context, network, true, reciprocate, reciprocateChance, nd));
123         }
124         
125         return network;
126     }

128     private static Network<BaseHuman> buildSFFoundation(Context context, Network network, NDParams nd)
129     {
130         IndexedIterable<Object> startNodes = context.getObjects(BaseHuman.class);
131         
132         //add random edges until everything has >=1 degree
133         boolean acceptableDegree = false;
134         
135         Iterable<BaseHuman> nodes = network.getNodes();
136         Iterator<BaseHuman> iter = nodes.iterator();
137         int minDegree = 9999999;
138         //get the min degree
139         while(iter.hasNext())
140         {
141             int currDegree = network.getDegree(iter.next());
142             if(currDegree < minDegree)
143                 minDegree = currDegree;
144         }
145         if(minDegree > 1)
146             acceptableDegree = true;
147         
148         while(!acceptableDegree)
149         {
150             //Generate src and dest until they're different
151             int source = (int) Math.round(Math.random() * (startNodes.size()));
152             int dest = (int) Math.round(Math.random() * (startNodes.size()));
153             
154             //Keep regenerating until they're different and within array bounds
155             while(source == dest || (source <= 0 || source > startNodes.size()) || (dest <= 0 || dest > startNodes.size()))
156             {
157                 source = (int) Math.round(Math.random() * (startNodes.size()));
158                 dest = (int) Math.round(Math.random() * startNodes.size());
159             }
160             
161             //check if we have NDParams, if not, no weights.
162             //System.out.println("Adding an edge...");
163             if(nd == null)
164                 network.addEdge((BaseHuman)startNodes.get(source-1), (BaseHuman)startNodes.get(dest-1));
165             else
166                 network.addEdge((BaseHuman)startNodes.get(source-1), 
167                         (BaseHuman)startNodes.get(dest-1), 
168                         Distributions.getND(nd));
169             //check for all degrees being >= 1
170             nodes = network.getNodes();
171             iter = nodes.iterator();
172             minDegree = 9999999;
173             //get the min degree
174             while(iter.hasNext())
175             {
176                 int currDegree = network.getDegree(iter.next());
177                 if(currDegree < minDegree)
178                     minDegree = currDegree;
179             }
180             if(minDegree > 1)
181                 acceptableDegree = true;
182         }
183         
184         return network;
185     }
