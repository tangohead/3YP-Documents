61      private static Graph<String, Long> SnowballSampler(Graph<String, Long> inGraph, Graph<String, Long> currGraph, List<String> vertexSet, int sampleSize, int prevSize)
62      {
63          Collection<Long> workingEdgeSet = new LinkedList<Long>(); 
64          Iterator<Long> edgeIter;
65          Pair<String> workingPair;
66          List<String> nextVertexSet = new ArrayList<String>();
67          
68          String newNode;
69          System.out.println("==============================================");
70          System.out.println("Graph size is " + currGraph.getVertexCount());
71          
72          
73          //Snowball sampler needs to pick randomly
74          //For each vertex...
75          
76          for(String o : vertexSet)
77          {
78              //...get the out-edges and iterate through them
79              Collection<Long> temp = inGraph.getOutEdges(o);
80              Iterator iter = temp.iterator();
81              while(iter.hasNext())
82              {
83                  workingEdgeSet.add((Long) iter.next());
84              }
85                  
86          }
87          
88          edgeIter = workingEdgeSet.iterator();
89          System.out.println("Current fringe is " + workingEdgeSet.size());
90          int maxSizeAtEnd = workingEdgeSet.size() + currGraph.getVertexCount();
91          boolean selective = false;
92          ArrayList<Integer> indexList = new ArrayList<Integer>();
93          System.out.println("Max size at end of iteration " + maxSizeAtEnd);
94          if( maxSizeAtEnd > sampleSize)
95          {
96              selective = true;
97              //Now we need to select just enough edges to meet the sample
98              int edgesNeeded = sampleSize - currGraph.getVertexCount();
99              int edgeCount = 0;
100             while(edgeCount < edgesNeeded)
101             {
102                 int index = (int) Math.round(Math.random() * workingEdgeSet.size());
103                 if(!indexList.contains(index))
104                 {
105                     indexList.add(index);
106                     edgeCount++;
107                 }
108             }
109             System.out.println("Needed " + edgesNeeded + " has " + edgeCount );
110                 
111         }
112         
113         int iterCount = 0;
114         while(edgeIter.hasNext())
115         {
116             //Find out which edge is not the one we are expanding from (i.e. the new node to be added)
117             Long currentEdge = edgeIter.next();
118             workingPair = inGraph.getEndpoints(currentEdge);
119             //if(workingPair.getFirst().equals(o))
120             //we got out edges, so it should be a second one    
121             newNode = workingPair.getSecond();
122             //else
123             //  newNode = workingPair.getFirst();
124             
125             //If it's not in the graph, add it
126             //note that we also check if we're being selective (on the final iteration), or not
127             //if we are, we look up the id to see if it's already in the graph
128             if(!currGraph.containsVertex(newNode) && ((selective && indexList.contains(iterCount)) || (!selective)))
129             {
130                 System.out.println("Are we selecting? " + selective + " At index? " + iterCount);
131                 currGraph.addVertex(newNode);
132                 nextVertexSet.add(newNode);
133             }
134             //If the edge to the new node doesn't exist, add it.
135             //  Note, this is separate because if we expand to a node that IS already in the graph but 
136             //  said node links to some other in-graph node, then we still need to add the edge.
137             if(!currGraph.containsEdge(currentEdge) && ((selective && indexList.contains(iterCount)) || (!selective)))
138                 currGraph.addEdge(currentEdge, workingPair, EdgeType.DIRECTED);
139             iterCount++;
140         }
141         
142         System.out.println("Graph size at end of iteration " + currGraph.getVertexCount());
143         //If the current graph is smaller than the sample size, and the previous graph size was less than current graph size (i.e. we
144         // are actually adding stuff), then recurse.
145         if(currGraph.getVertexCount() < sampleSize  && (prevSize < currGraph.getVertexCount()))
146             currGraph = SnowballSampler(inGraph, currGraph, nextVertexSet, sampleSize, currGraph.getVertexCount());
147         
148         return currGraph;
149     }
