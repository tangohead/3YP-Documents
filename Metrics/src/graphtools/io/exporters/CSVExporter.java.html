1   package graphtools.io.exporters;
2   
3   import java.io.FileWriter;
4   import java.io.IOException;
5   import java.lang.reflect.Field;
6   import java.util.ArrayList;
7   import java.util.List;
8   
9   import agents.BaseHuman;
10  
11  public class CSVExporter {
12  
13      public static void exportToCSV(Iterable<BaseHuman> nodes, String filename)
14      {
15          //System.out.println("Writing to CSV - " + filename);
16          Field[] nodeFields = BaseHuman.class.getDeclaredFields();
17          FileWriter fw;
18  
19          ArrayList<BaseHuman> humanNodes = new ArrayList<BaseHuman>();
20          for(Object o : nodes)
21          {
22              if(o instanceof BaseHuman)
23                  humanNodes.add((BaseHuman) o);
24              
25          }
26          try {
27               fw = new FileWriter(filename + ".csv");
28               String writeLine = "";
29                  
30              for(Field f : nodeFields)
31              {
32                  if(f.getName() != "context" && f.getName() != "network")
33                  {
34                      writeLine += f.getName() + ",";
35                      //bit of a hack, allows the next bit to be as general as possible
36                      f.setAccessible(true);
37                  }
38                  
39              }
40              //Remove final comma
41              writeLine = writeLine.substring(0, writeLine.length()-1);
42              writeLine += "\n";
43              fw.append(writeLine);
44              for(BaseHuman b : humanNodes)
45              {
46                  writeLine ="";
47                  for(Field f : nodeFields)
48                  {
49                      if(f.getName() != "context" && f.getName() != "network")
50                      {
51                          Object o = f.get(b);
52                          Object tempVal = null;
53  
54                          if(o instanceof Integer)
55                              tempVal = (Integer) o;
56                          else if (o instanceof Long)
57                              tempVal = (Long) o;
58                          else if (o instanceof Double)
59                              tempVal = (Double) o;
60                          else if(o instanceof Boolean)
61                              tempVal = (Boolean) o;
62                          else if(o instanceof String)
63                              tempVal = (String) o;
64                          writeLine += tempVal + ",";
65                      }
66                  }
67                  writeLine = writeLine.substring(0, writeLine.length()-1);
68                  writeLine += "\n";
69                  fw.append(writeLine);
70              }
71              
72              fw.close();
73              //System.out.println("CSV Created.");
74          } catch (IOException e) {
75              // TODO Auto-generated catch block
76              System.out.println("*********** ERROR: File writer could not be created ***********");
77              return;
78          } catch (IllegalArgumentException e) {
79              // TODO Auto-generated catch block
80              e.printStackTrace();
81          } catch (IllegalAccessException e) {
82              // TODO Auto-generated catch block
83              e.printStackTrace();
84          }
85          
86          
87      }
88  }
