31      /**
32       * Reads in a GraphML file and converts it to a Repast network
33       * @param filename
34       * @param graphName
35       * @param context
36       * @return
37       */
38      public static RepastSummary GraphMLToRepast(String filename, String graphName, Context context, boolean loadNetworkOnly, boolean loadAttrsOnly)
39      {
40          try {
41              File file = new File(filename);
42              JAXBContext jaxbContext = JAXBContext.newInstance(GraphMLObject.class);
43       
44              //Read in GraphML
45              Unmarshaller jaxbUnmarshaller = jaxbContext.createUnmarshaller();
46              GraphMLObject GML = (GraphMLObject) jaxbUnmarshaller.unmarshal(file);
47              
48              //This is a way to return both a context and a network
49              RepastSummary rs = new RepastSummary();
50  
51              //Build the temporary network
52              NetworkBuilder<BaseHuman> builder = new NetworkBuilder(graphName, context, true);
53              Network<BaseHuman> tempNetwork = builder.buildNetwork();
54              
55              //We want to make it easier to look up values, so we get this 
56              // map of class var names to ids
57              HashMap<String, String> nodeKeyMap = GML.getNodeKeyRegister();
58              HashMap<String, String> edgeKeyMap = GML.getEdgeKeyRegister();
59              
60              //We know we only have weights for edges on import, so
61              String edgeWeightID = edgeKeyMap.get("edgeWeight");
62              Set<String> s = nodeKeyMap.keySet();
63              Iterator<String> si = s.iterator();
64              while(si.hasNext())
65              {
66                  String k = si.next();
67                  System.out.println(k + " " + nodeKeyMap.get(k));
68              }
69              for(XMLNode node : GML.getXMLNodes())
70              {
71  //              Collection<String> c = nodeKeyMap.values();
72  //              Iterator<String> i = c.iterator();
73  //              while(i.hasNext())
74  //                  System.out.println(i.next());
75                  if(loadNetworkOnly)
76                  {
77                      NDParams nd = new NDParams(0.8, 0.4, 0, 1);
78                      context.add(new BaseHuman(node.getID(), context, tempNetwork, false, false, 0, nd));
79                  }
80                  else if(loadAttrsOnly)
81                  {
82                      context.add(new BaseHuman(context, node.getID(), node.getAttrList(), nodeKeyMap, true));
83                  }
84                  else
85                      context.add(new BaseHuman(context, node.getID(), node.getAttrList(), nodeKeyMap, false));
86              }
87              
88              //now generate a hashmap of objects to look up when adding edges
89              IndexedIterable<BaseHuman> index = context.getObjects(BaseHuman.class);
90              Iterator<BaseHuman> iter = index.iterator();
91              HashMap<String, BaseHuman> objMap = new HashMap<String, BaseHuman>();
92              
93              while(iter.hasNext())
94              {
95                  BaseHuman tmp = iter.next();
96                  objMap.put(tmp.getID(), tmp);
97                  
98              }
99              //Add the edges
100             for(Edge edge : GML.getEdges())
101             {
102                 HashMap<String, String> attrList = edge.getAttrList();
103                 if(attrList.get(edgeWeightID) == null)
104                     tempNetwork.addEdge(objMap.get(edge.getSource()), objMap.get(edge.getTarget()), Distributions.getNDWithLimits(0.5, 0.3, 0, 1));
105                 else
106                     tempNetwork.addEdge(objMap.get(edge.getSource()), objMap.get(edge.getTarget()), Double.parseDouble(attrList.get(edgeWeightID)));
107             }
108             
109             rs.setContext(context);
110             rs.setNetwork(tempNetwork);
111             //go through context and assign network
112             index = context.getObjects(BaseHuman.class);
113             iter = index.iterator();
114 
115             while(iter.hasNext())
116             {
117                 BaseHuman tmp = iter.next();
118                 tmp.setNetwork(tempNetwork);
119             }
120             
121             
122             return rs;
123           } catch (JAXBException e) {
124             e.printStackTrace();
125           }
126         return null;
127     }
