23      private String filePath;

29      public StanfordParser(String filePath)
30      {
31          this.filePath = filePath;
32      }
33      
34      public boolean Parse() 
35      {
36          boolean error = false;
37          try
38          {
39              FileInputStream fstream = new FileInputStream(filePath);
40              DataInputStream in = new DataInputStream(fstream);
41              BufferedReader br = new BufferedReader(new InputStreamReader(in));
42              String strLine;
43              String tempArr[];
44              
45              
46              //Code from http://www.roseindia.net/java/beginners/java-write-to-file.shtml
47              while ((strLine = br.readLine()) != null)   {
48                  //System.out.println(strLine);
49                  if(strLine.startsWith("#") == true )
50                  {
51                      fileComments += strLine + "\n";
52                  }
53                  else
54                  {
55                      tempArr = strLine.split("\t");
56                      String insertArr[] = new String[2];
57                      for(int i = 0; i < insertArr.length; i++)
58                      {
59                          insertArr[i] = tempArr[i];
60                      }
61                      //System.out.println("Adding " + insertArr[0] + " to " + insertArr[1]);
62                      edges.add(insertArr);
63                  }
64              }
65              in.close();
66          }
67          catch (FileNotFoundException e){
68              System.err.println("File not found!");
69              error = true;
70          } catch (IOException e) {
71              // TODO Auto-generated catch block
72              System.err.println("I/O Error!");
73              error = true;
74          }
75          return error;
76      }
77      
78      public ArrayList<String[]> getDataSet()
79      {
80          return edges;
81      }
82      
83      public String getComments() 
84      {
85          return fileComments;
86      }

88      public Graph<String, Long> toGraph()
89      {
90          //Add vertices to graph first
91          //STORE ADDED IDS IN COLLECTION
92          //CHECK FOR ID BEFORE ADDING NEW NODE
93          HashSet<String> nodes = new HashSet<String>();
94          Long edgeName = (long) 0;
95          EdgeType edgeType = EdgeType.DIRECTED;
96          Pair<String> vPair;
97          for(int i = 0; i < edges.size(); i++)
98          {
99              //System.out.println("Adding " + edges.get(i)[0] + " to " + edges.get(i)[1]);
100             
101             if(!nodes.contains(edges.get(i)[0]))
102             {
103                 graphRep.addVertex(edges.get(i)[0]);
104                 nodes.add(edges.get(i)[0]);
105                 //System.out.println(graphRep.getVertexCount());
106             }
107             if(!nodes.contains(edges.get(i)[1]))
108             {
109                 graphRep.addVertex(edges.get(i)[1]);
110                 nodes.add(edges.get(i)[1]);
111                 //System.out.println(graphRep.getVertexCount());
112             }
113             vPair = new Pair<String>(edges.get(i)[0], edges.get(i)[1]);
114             edgeName++;
115             graphRep.addEdge(edgeName, vPair, edgeType);
116             //System.out.println(graphRep.getVertexCount());
117         }
118         
119         return graphRep;
120     }
