99  class FileStore
100 {
101     String name;
102     String header;
103     ArrayList<ArrayList<String>> columns = new ArrayList<ArrayList<String>>();
104     
105     FileStore(String name)
106     {
107         this.name = name;
108     }
109     
110     public void addCol(String colName)
111     {
112         header += colName + ",";
113         columns.add(new ArrayList<String>());
114     }
115     
116     public void addRowVal(String colName, int columnNumber, String value)
117     {
118         try {
119             columns.get(columnNumber).add(value);
120         } 
121         catch(IndexOutOfBoundsException e)
122         {
123             header += colName + ",";
124             columns.add(new ArrayList<String>());
125             columns.get(columnNumber).add(value);
126         }
127         
128     }
129     
130     public void writeToFile(String dir)
131     {
132         header = header.substring(0, header.length() - 1);
133         try {
134             FileWriter fw = new FileWriter(dir + name + ".csv");
135             fw.write(header + "\n");
136             
137             for(int i = 0; i < columns.get(0).size(); i++)
138             {
139                 String line = "";
140                 for(int j = 0; j < columns.size(); j++)
141                 {
142                     line += columns.get(j).get(i) + ",";
143                 }
144                 line = line.substring(0, line.length() - 1);
145                 fw.write(line + "\n");
146             }
147             
148             fw.close();
149         } catch (IOException e) {
150             // TODO Auto-generated catch block
151             e.printStackTrace();
152         }
153         
154         
155     }
156     
157     public void printCol()
158     {
159         for(ArrayList<String> col : columns)
160         {
161             System.out.println(col.size() + " " );
162         }
163     }
164 }
